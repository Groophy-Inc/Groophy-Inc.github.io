        /// <summary>
        /// https://groophy-inc.github.io/lim
        /// </summary>
        public enum Infinity
        {
            Positive,
            Negative
        }

        /// <summary>
        /// https://groophy-inc.github.io/lim
        /// </summary>
        /// <param name="x"></param>
        /// <param name="formula"></param>
        /// <returns></returns>
        public static GValue Lim(GValue x, string formula)
        {
            MathParser parser = new MathParser();
            parser.LocalVariables.Add("x", x.get_Double);
            return new GValue(parser.Parse(formula).ToString());
        }

        /// <summary>
        /// https://groophy-inc.github.io/lim
        /// </summary>
        /// <param name="x"></param>
        /// <param name="formula"></param>
        /// <returns></returns>
        public static GValue Lim(string x_, string formula)
        {
            GValue x = new GValue(x_);
            MathParser parser = new MathParser();
            parser.LocalVariables.Add("x", x.get_Double);
            return new GValue(parser.Parse(formula).ToString());
        }

        /// <summary>
        /// https://groophy-inc.github.io/lim
        /// </summary>
        /// <param name="x"></param>
        /// <param name="formula"></param>
        /// <returns></returns>
        public static GValue Lim(Infinity x, string formula)
        {
            MathParser parser = new MathParser();
            StringBuilder sb = new StringBuilder();
            if (x == Infinity.Positive)
            {
                parser.LocalVariables.Add("x", 0);
                sb.Append(Environment.NewLine+"0: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = 1;
                sb.Append("1: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = 2;
                sb.Append("2: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = 3;
                sb.Append("3: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = 4;
                sb.Append("4: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = 5;
                sb.Append("5: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = 10;
                sb.Append("10: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = 20;
                sb.Append("20: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = 100;
                sb.Append("100: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = 5000;
                sb.Append("5000: " + parser.Parse(formula).ToString() + Environment.NewLine + Environment.NewLine);

                double f5 = 0;
                for (int i = 0;i < 5;i++)
                {
                    parser.LocalVariables["x"] = i;
                    f5 += parser.Parse(formula);
                }
                f5 /= 5;

                double f20 = 0;
                for (int i = 5; i < 20; i++)
                {
                    parser.LocalVariables["x"] = i;
                    f20 += parser.Parse(formula);
                }
                f20 /= 15;

                double f100 = 0;
                for (int i = 20; i < 100; i++)
                {
                    parser.LocalVariables["x"] = i;
                    f100 += parser.Parse(formula);
                }
                f100 /= 80;

                sb.Append("0~5 average: " + f5.ToString() + Environment.NewLine);
                sb.Append("5~20 average: " + f20.ToString() + Environment.NewLine);
                sb.Append("20~100 average: " + f100.ToString() + Environment.NewLine);
            }
            else if (x == Infinity.Negative)
            {
                parser.LocalVariables.Add("x", 0);
                sb.Append(Environment.NewLine+"0: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = -1;
                sb.Append("-1: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = -2;
                sb.Append("-2: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = -3;
                sb.Append("-3: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = -4;
                sb.Append("-4: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = -5;
                sb.Append("-5: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = -10;
                sb.Append("-10: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = -20;
                sb.Append("-20: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = -100;
                sb.Append("-100: " + parser.Parse(formula).ToString() + Environment.NewLine);
                parser.LocalVariables["x"] = -5000;
                sb.Append("-5000: " + parser.Parse(formula).ToString() + Environment.NewLine + Environment.NewLine);

                double f5 = 0;
                for (int i = 0; i > -5; i--)
                {
                    parser.LocalVariables["x"] = i;
                    f5 += parser.Parse(formula);
                }
                f5 /= 5;

                double f20 = 0;
                for (int i = -5; i > -20; i--)
                {
                    parser.LocalVariables["x"] = i;
                    f20 += parser.Parse(formula);
                }
                f20 /= 15;

                double f100 = 0;
                for (int i = -20; i > -100; i--)
                {
                    parser.LocalVariables["x"] = i;
                    f100 += parser.Parse(formula);
                }
                f100 /= 80;

                sb.Append("-0~-5 average: " + f5.ToString() + Environment.NewLine);
                sb.Append("-5~-20 average: " + f20.ToString() + Environment.NewLine);
                sb.Append("-20~-100 average: " + f100.ToString() + Environment.NewLine);
            }
            GValue g = new GValue("0");
            g.get_Tag = sb.ToString();
            return g;
        }
